 11 /* URL do servico AcaaS. */
 12 var URL_ACAAS = "http://acaas-robsonmartins.rhcloud.com/acaas";
 13 
 14 var STATUS_LISTEN  = "Fale ao microfone...";
 15 var STATUS_PROCESS = "Analisando...";
 16 
 17 var videoPlayer = document.getElementById("video_player");
 18 var statusText = document.getElementById("status_text");
 19 var outputText = document.getElementById("output");
 20 var btnStart    = document.getElementById("btn_start");
 21 var btnStop     = document.getElementById("btn_stop");
 22 
 23 var chkBrws = new CheckBrowser();
 24 var acaas = new Acaas(URL_ACAAS);
 25 var speech = new Speech();
 26 
 27 var audioMediaType = "audio/wav";
 28 var videoMediaType = "video/ogg";
 29 
 30 var videoBuffer = null;
 31 var audioBuffer = null;
 32 
 33 var videoLoaded = false;
 34 var searching = false;
 35 var started = false;
 36 var poster = null;
 37 
 38 window.URL = window.URL || window.webkitURL;
 39 
 40 videoPlayer.addEventListener("ended",getNextVideo);
 41 
 42 if (chkBrws.hasAudioVideo() && chkBrws.hasRemoteRequest() &&
 43 		chkBrws.hasAudioCapture()) {
 44 	btnStart.disabled = false;
 45 } else {
 46 	window.alert("Navegador n\u00E3o suportado.");
 47 }
 48 
 49 /** @private */
 50 function start() {
 51 	btnStart.disabled = true;
 52 	btnStop.disabled  = false;
 53 	if (poster == null) {
 54 		poster = videoPlayer.poster;
 55 	}
 56 	if (videoPlayer.canPlayType("video/ogg") == "") {
 57 		videoMediaType = "video/mp4";
 58 	} else {
 59 		videoMediaType = "video/ogg";
 60 	}
 61 	outputText.value = "";
 62 	statusText.innerHTML = STATUS_LISTEN;
 63 	audioBuffer = new Queue();
 64 	videoBuffer = new Queue();
 65 	started = true;
 66 	speech.startCapture(onAudioCapture, onAudioError);
 67 }
 68 
 69 /** @private */
 70 function stop() {
 71 	if (!started) { return; }
 72 	btnStart.disabled = false;
 73 	btnStop.disabled  = true;
 74 	started = false;
 75 	speech.stopCapture();
 76 	setVideoPoster();
 77 	statusText.innerHTML = "";
 78 	outputText.value = "";
 79 	audioBuffer = null;
 80 	videoBuffer = null;
 81 }
 82 
 83 /** @private */
 84 function onAudioCapture(audio) {
 85 	if (!started) { return; }
 86 	audioBuffer.enqueue(audio);
 87 	querySTT();
 88 }
 89 
 90 /** @private */
 91 function querySTT() {
 92 	if (!started || searching) { return; }
 93 	var audio = audioBuffer.dequeue();
 94 	if (audio != undefined) {
 95 		statusText.innerHTML = STATUS_PROCESS;
 96 		videoLoaded = false;
 97 		searching = true;
 98 		acaas.getSTT(audio, audioMediaType, 
 99 				onMediaSTT, onFinishMediaSTT, onErrorSTT);
100 	} else {
101 		statusText.innerHTML = STATUS_LISTEN;
102 	}
103 }
104 
105 /** @private */
106 function onMediaSTT(data) {
107 	if (!started) { return; }
108 	var text = ""; 
109 	if (data != null && data.items != null && data.items.length != 0) {
110 		for (var i = 0; i < data.items.length; i++) {
111 			text += data.items[i] + " ";
112 		}
113 		outputText.value += text;
114 		acaas.getT2Libras(text, videoMediaType, 
115 				onMediaT2Libras, onFinishMediaT2Libras, onErrorT2Libras);
116 	}
117 }
118 
119 /** @private */
120 function onFinishMediaSTT() {
121 	onFinishMediaT2Libras();
122 	querySTT();
123 }
124 
125 /** @private */
126 function onAudioError(e) {
127 	statusText.innerHTML = "Erro ao obter audio: " + e.message;
128 }
129 
130 /** @private */
131 function onErrorSTT(status, text) {
132 	onFinishMediaSTT();
133 }
134 
135 /** @private */
136 function onErrorT2Libras(status, text) {
137 	onFinishMediaT2Libras();
138 }
139 
140 /** @private */
141 function onMediaT2Libras(media, info) {
142 	if (!started) { return; }
143 	videoBuffer.enqueue({"media": media, "info": info});
144 	if (!videoLoaded) {
145 		videoLoaded = true;
146 		videoPlayer.poster = "";
147 		getNextVideo();
148 	}
149 }
150 
151 /** @private */
152 function onFinishMediaT2Libras() {
153 	if (!started) { return; }
154 	statusText.innerHTML = STATUS_LISTEN;
155 	videoLoaded = false;
156 	searching = false;
157 }
158 
159 /** @private */
160 function getNextVideo() {
161 	if (!started) { return; }
162 	var data = videoBuffer.dequeue();
163 	if (data != undefined) {
164 		window.URL.revokeObjectURL(videoPlayer.src);
165 		videoPlayer.src = window.URL.createObjectURL(data.media);
166 		videoPlayer.type = videoMediaType;
167 		videoPlayer.load();
168 		videoPlayer.play();
169 	} else if (videoLoaded){
170 		setTimeOut(getNextVideo,200);
171 	} else {
172 		setVideoPoster();
173 	}
174 }
175 
176 /** @private */
177 function setVideoPoster() {
178 	videoPlayer.poster = "#";
179 	window.URL.revokeObjectURL(videoPlayer.src);
180 	videoPlayer.src = "#";
181 	videoPlayer.poster = poster;
182 }
