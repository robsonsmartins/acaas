  9 /**
 10  * @class Implementa metodos para verificar features do browser.
 11  * @constructor
 12  * @description Cria um objeto para verificacao de features do browser.
 13  */
 14 function CheckBrowser() {
 15 	/**
 16 	 * @memberOf CheckBrowser 
 17 	 * @function
 18 	 * @description Verifica se browser suporta features para
 19 	 *   requisicao HTTP remota (XMLHttpRequest, Blob, window.URL).
 20 	 * @returns True se browser suporta todos os recursos necessarios,
 21 	 *   false se nao.
 22 	 */
 23 	this.hasRemoteRequest = function() {
 24 		window.URL = window.URL || window.webkitURL;
 25 		var hasXHR  = ((typeof XDomainRequest != "undefined") ||
 26 		               (typeof XMLHttpRequest != "undefined"));
 27 		var hasBlob = (typeof Blob != "undefined");
 28 		var hasWindowURL = (window.URL != null && 
 29 							window.URL != "undefined");
 30 		return (hasXHR && hasBlob && hasWindowURL);
 31 	};
 32 	
 33 	/**
 34 	 * @memberOf CheckBrowser 
 35 	 * @function
 36 	 * @description Verifica se browser suporta features para
 37 	 *   reproducao de midia (Video, Audio).
 38 	 * @returns True se browser suporta todos os recursos necessarios,
 39 	 *   false se nao.
 40 	 */
 41 	this.hasAudioVideo = function() {
 42 		var hasVideo = !!document.createElement('video').canPlayType;
 43 		var hasAudio = !!document.createElement('audio').canPlayType;
 44 		return (hasVideo &&	hasAudio);
 45 	};
 46 	
 47 	/**
 48 	 * @memberOf CheckBrowser 
 49 	 * @function
 50 	 * @description Verifica se browser suporta features para
 51 	 *   captura de audio (getUserMedia, AudioContext, DataView,
 52 	 *   ArrayBuffer, Blob).
 53 	 * @returns True se browser suporta todos os recursos necessarios,
 54 	 *   false se nao.
 55 	 */
 56 	this.hasAudioCapture = function() {
 57 		window.AudioContext = 
 58 			window.AudioContext || window.webkitAudioContext ||
 59 			window.mozAudioContext;
 60 		navigator.getUserMedia = 
 61 			navigator.getUserMedia || navigator.webkitGetUserMedia ||
 62 	    	navigator.mozGetUserMedia || navigator.msGetUserMedia;
 63 		var hasBlob = (typeof Blob != "undefined");
 64 		var hasArrayBuf = (typeof ArrayBuffer != "undefined");
 65 		var hasDataView = (typeof DataView != "undefined");
 66 		var hasAudioContext = (window.AudioContext != null && 
 67 				window.AudioContext != "undefined");
 68 		var hasGetUserMedia = (navigator.getUserMedia != null && 
 69 				navigator.getUserMedia != "undefined");
 70 		return (hasBlob && hasArrayBuf && hasDataView && 
 71 				hasAudioContext && hasGetUserMedia);
 72 	};
 73 }
