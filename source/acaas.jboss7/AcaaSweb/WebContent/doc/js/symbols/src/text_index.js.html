 11 /* URL do servico AcaaS. */
 12 var URL_ACAAS = "http://acaas-robsonmartins.rhcloud.com/acaas";
 13 
 14 var audioPlayer = document.getElementById("audio_player");
 15 var videoPlayer = document.getElementById("video_player");
 16 var input       = document.getElementById("input");
 17 var btnStart    = document.getElementById("btn_start");
 18 
 19 var chkBrws = new CheckBrowser();
 20 var acaas = new Acaas(URL_ACAAS);
 21 
 22 var audioMediaType = "audio/wav";
 23 var videoMediaType = "video/ogg";
 24 
 25 var videoBuffer = null;
 26 var audioBuffer = null;
 27 
 28 var audioLoaded = false;
 29 var videoLoaded = false;
 30 
 31 var poster = null;
 32 
 33 window.URL = window.URL || window.webkitURL;
 34 
 35 audioPlayer.addEventListener("ended",getNextAudio);
 36 videoPlayer.addEventListener("ended",getNextVideo);
 37 input.addEventListener("keyup",inputKeyPress);
 38 
 39 if (chkBrws.hasAudioVideo() && chkBrws.hasRemoteRequest()) {
 40 	input.readOnly = false;
 41 } else {
 42 	window.alert("Navegador n\u00E3o suportado.");
 43 }
 44 
 45 /** @private */
 46 function inputKeyPress(event) {
 47 	btnStart.disabled = (input.value == null || input.value == "");
 48 }
 49 
 50 /** @private */
 51 function start() {
 52 	btnStart.disabled = true;
 53 	input.readOnly = true;
 54 	if (poster == null) {
 55 		poster = videoPlayer.poster;
 56 	}
 57 	audioLoaded = false;
 58 	videoLoaded = false;
 59 	if (videoPlayer.canPlayType("video/ogg") == "") {
 60 		videoMediaType = "video/mp4";
 61 	} else {
 62 		videoMediaType = "video/ogg";
 63 	}
 64 	videoBuffer = new Queue();
 65 	audioBuffer = new Queue();
 66 	acaas.getTTS(input.value, audioMediaType, onMediaTTS, onFinishMediaTTS);
 67 	acaas.getT2Libras(input.value, videoMediaType, onMediaT2Libras, onFinishMediaT2Libras);
 68 }
 69 
 70 /** @private */
 71 function onMediaTTS(media, info) {
 72 	audioBuffer.enqueue({"media": media, "info": info});
 73 	if (!audioLoaded) {
 74 		audioLoaded = true;
 75 		getNextAudio();
 76 	}
 77 }
 78 
 79 /** @private */
 80 function onFinishMediaTTS() {
 81 	audioLoaded = false;
 82 	btnStart.disabled = videoLoaded;
 83 	input.readOnly = btnStart.disabled;
 84 }
 85 
 86 /** @private */
 87 function onMediaT2Libras(media, info) {
 88 	videoBuffer.enqueue({"media": media, "info": info});
 89 	if (!videoLoaded) {
 90 		videoLoaded = true;
 91 		videoPlayer.poster = "";
 92 		getNextVideo();
 93 	}
 94 }
 95 
 96 /** @private */
 97 function onFinishMediaT2Libras() {
 98 	videoLoaded = false;
 99 	btnStart.disabled = audioLoaded;
100 	input.readOnly = btnStart.disabled;
101 }
102 
103 /** @private */
104 function getNextAudio() {
105 	var data = audioBuffer.dequeue();
106 	if (data != undefined) {
107 		window.URL.revokeObjectURL(audioPlayer.src);
108 		audioPlayer.src = window.URL.createObjectURL(data.media);
109 		audioPlayer.type = audioMediaType;
110 		audioPlayer.load();
111 		audioPlayer.play();
112 	} else if (audioLoaded){
113 		setTimeOut(getNextAudio,200);
114 	}
115 }
116 
117 /** @private */
118 function getNextVideo() {
119 	var data = videoBuffer.dequeue();
120 	if (data != undefined) {
121 		window.URL.revokeObjectURL(videoPlayer.src);
122 		videoPlayer.src = window.URL.createObjectURL(data.media);
123 		videoPlayer.type = videoMediaType;
124 		videoPlayer.load();
125 		videoPlayer.play();
126 	} else if (videoLoaded){
127 		setTimeOut(getNextVideo,200);
128 	} else {
129 		videoPlayer.poster = "#";
130 		window.URL.revokeObjectURL(videoPlayer.src);
131 		videoPlayer.src = "#";
132 		videoPlayer.poster = poster;
133 	}
134 }
