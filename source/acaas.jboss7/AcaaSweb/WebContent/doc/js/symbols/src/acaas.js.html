101 /**
102  * @class Implementa um consumidor de um Servico JSON do AcaaS.
103  * @constructor
104  * @description Cria um objeto para acessar servicos JSON do AcaaS.
105  * @param {String} urlService URL do Servico AcaaS. 
106  */
107 function AcaasService(urlService) {
108 	var acaasFormatParam = "format";
109 	var acaas = new AcaasConnection();
110 
111 	/**
112 	 * @memberOf AcaasService 
113 	 * @function
114 	 * @description Consulta um servico do AcaaS e retorna a resposta
115 	 *   como um Objeto JSON.
116 	 * @param {Raw} data Dados para consultar.
117 	 * @param {String} mediaType Tipo MIME do formato dos dados em <em>data</em>.
118 	 * @param {function(JsonObject data)} [onData] Callback do evento
119 	 *   disparado quando os dados de uma midia estao disponiveis.
120 	 * @param {function()} [onFinish] Callback do evento disparado ao final
121 	 *   da traducao.
122 	 * @param {function(Number status, String statusText)} [onError] Callback
123 	 *   do evento disparado quando a requisicao resultou em erro.
124 	 */
125 	this.get = function (data, mediaType, onData, onFinish, onError) {
126 		var url = urlService + "?" + acaasFormatParam + "=" + mediaType;
127 		acaas.queryService(url, data, true,
128 				function(json) { 
129 					if (onData) { onData(json); }
130 					if (onFinish) { onFinish(); }
131 				},
132 				onError
133 		);		
134 	};
135 }

138 /**
139  * @class Implementa um consumidor de um Servico de Midia do AcaaS.
140  * @constructor
141  * @description Cria um objeto para acessar servicos de Midia do AcaaS.
142  * @param {String} urlService URL do Servico AcaaS. 
143  */
144 function AcaasMediaService(urlService) {
145 	var urlMedia = urlService + "/media"; 
146 	var acaasFormatParam = "format";
147 	var acaasIformatParam = "iformat";
148 	var acaasOformatParam = "oformat";
149 	var mediaCount = -1;
150 	var jsonData = null;
151 	var acaas = new AcaasConnection();
152 
153 	/**
154 	 * @memberOf AcaasMediaService 
155 	 * @function
156 	 * @description Consulta um servico do AcaaS e retorna a resposta de
157 	 *   midia como um Blob.
158 	 * @param {Raw} [data] Dados para consultar.
159 	 * @param {String} [iMediaType] Tipo MIME do formato dos dados em <em>data</em>.
160 	 * @param {String} [oMediaType] Tipo MIME do formato desejado para a midia.
161 	 * @param {function(Blob data, JsonObject info)} [onData] Callback do evento
162 	 *   disparado quando os dados de uma midia estao disponiveis.
163 	 * @param {function()} [onFinish] Callback do evento disparado ao final
164 	 *   da traducao.
165 	 * @param {function(Number status, String statusText)} [onError] Callback
166 	 *   do evento disparado quando a requisicao resultou em erro.
167 	 */
168 	this.get = function (data, iMediaType, oMediaType, onData, onFinish, onError) {
169 		var url = urlService;
170 		var mediaType = null;
171 		if (iMediaType != null && oMediaType != null) {
172 			url += "?" + acaasIformatParam + "=" + iMediaType + "&" + 
173 						 acaasOformatParam + "=" + oMediaType;
174 			mediaType = oMediaType;
175 		} else if (iMediaType != null) {
176 			url += "?" + acaasFormatParam + "=" + iMediaType;
177 			mediaType = iMediaType;
178 		} else if (oMediaType != null) {
179 			url += "?" + acaasFormatParam + "=" + oMediaType;
180 			mediaType = oMediaType;
181 		}
182 		acaas.queryService(url, data, true,
183 				function(json) { 
184 					jsonData = json; 
185 					mediaCount = -1; 
186 					getMediaData(mediaType, onData, onFinish, onError); 
187 				},
188 				onError
189 		);		
190 	};
191 	
192 	/** @private */
193 	function getMediaData(mediaType, onData, onFinish, onError) {
194 		if (jsonData != null && jsonData.items != null && 
195 				mediaCount < jsonData.items.length - 1) {
196 			mediaCount++;
197 			var url = urlMedia + "/" + jsonData.items[mediaCount].media;
198 			var info = jsonData.items[mediaCount];
199 			acaas.getMedia(url, info, mediaType, true,
200 					function(mediaData, mediaInfo) {
201 						processMedia(mediaData, mediaInfo, mediaType, onData, onFinish, onError);	
202 					},
203 					onError
204 			);
205 		} else {
206 			if (onFinish) { onFinish(); }
207 		}
208 	}
209 	
210 	/** @private */
211 	function processMedia(data, info, mediaType, onData, onFinish, onError) {
212 		if (onData) { onData(data, info); }
213 		getMediaData(mediaType, onData, onFinish, onError);
214 	}
215 }

219  * @class Implementa conexao HTTP a um Servico AcaaS.

221  * @description Cria um objeto para conectar a um servico AcaaS.
222  */
223 function AcaasConnection() {
224 	/**
225 	 * @memberOf AcaasConnection 
226 	 * @function
227 	 * @description Consulta um servico do AcaaS (via HTTP POST),
228 	 *   e retorna a resposta como um Objeto JSON.
229 	 * @param {String} url URL do Servico AcaaS,
230 	 *   contendo todos parametros necessarios.
231 	 * @param {Raw} data Dados para consultar (corpo da requisicao HTTP POST).
232 	 * @param {boolean} [async] Indica se a requisicao sera' assincrona.
233 	 * @param {function(JsonObject data)} [onData] Callback do evento
234 	 *   disparado quando os dados de uma midia estao disponiveis.
235 	 * @param {function(Number status, String statusText)} [onError] Callback
236 	 *   do evento disparado quando a requisicao resultou em erro.
237 	 */
238 	this.queryService = function (url, data, async, onData, onError) {
239 		if (async == null) { async = false; }
240 		var xhttp = new XMLHttpRequest();
241 		xhttp.open('POST', url, async);
242 		/** @private */
243 		xhttp.onload = function(event) {
244 			var jsonStr = this.response;
245 			var json = null;
246 			if (jsonStr != null) { json = JSON.parse(jsonStr); }
247 			if (onData) { onData(json); }
248 		}; 
249 		/** @private */
250 		xhttp.onerror = function(event) {
251 			var reqStatus = this.status; 
252 			var reqStatusText = this.statusText;
253 			if (onError) { onError(reqStatus, reqStatusText); }
254 		};
255 		xhttp.send(data);
256 	};
257 	
258 	/**
259 	 * @memberOf AcaasConnection 
260 	 * @function
261 	 * @description Consulta um servico de midia do AcaaS (via HTTP GET),
262 	 *   e retorna a resposta como um Blob.
263 	 * @param {String} url URL do Servico AcaaS,
264 	 *   contendo todos parametros necessarios.
265 	 * @param {JsonObject} [info] Objeto JSON com informacoes sobre a midia
266 	 *   requisitada.
267 	 * @param {String} mediaType Tipo MIME do formato desejado para a midia.
268 	 * @param {boolean} [async] Indica se a requisicao sera' assincrona.
269 	 * @param {function(Blob data, JsonObject info)} [onData] Callback do evento
270 	 *   disparado quando os dados de uma midia estao disponiveis.
271 	 * @param {function(Number status, String statusText)} [onError] Callback
272 	 *   do evento disparado quando a requisicao resultou em erro.
273 	 */
274 	this.getMedia = function (url, info, mediaType, async, onData, onError) {
275 		if (async == null) { async = false; }
276 		var xhttp = new XMLHttpRequest();
277 		if ("withCredentials" in xhttp) {
278 			xhttp.open('GET', url, async);
279 		} else if (typeof XDomainRequest != "undefined") {
280 			xhttp = new XDomainRequest();
281 			xhttp.open('GET', url);
282 		}
283 		xhttp.onload = function(event) {
284 			var blob = null;
285 			try {
286 				blob = new Blob([this.response], {type: mediaType});
287 			} catch(e) {
288 				var blobBuilder = null;
289 				if (typeof BlobBuilder != "undefined") {
290 					blobBuilder = new BlobBuilder();
291 				} else if (typeof WebKitBlobBuilder != "undefined") {
292 					blobBuilder = new WebKitBlobBuilder();
293 				} else if (typeof MozBlobBuilder != "undefined") {
294 					blobBuilder = new MozBlobBuilder();
295 				} else if (typeof MSBlobBuilder != "undefined") {
296 					blobBuilder = new MSBlobBuilder();
297 				}
298 				blobBuilder.append(this.response);
299 				blob = blobBuilder.getBlob(mediaType);
300 			}
301 			if (onData) { onData(blob, info); }
302 		};
303 		/** @private */
304 		xhttp.onerror = function(event) {
305 			var reqStatus = this.status; 
306 			var reqStatusText = this.statusText;
307 			if (onError) { onError(reqStatus, reqStatusText); }
308 		};
309 		try {
310 			xhttp.responseType = "arraybuffer";
311 			xhttp.send();
312 		} catch(e) {
313 			xhttp.responseType = "";
314 			xhttp.send();
315 		}
316 	};
317 }

580 // null.
581 
582         case 'object':
583

585 // so watch out for that case.
586 
587             if (!value) {
588                 return 'null';
589             }

591 // Make an array to hold the partial results of stringifying this object value.
592 
593             gap += indent;
594             partial = [];
595 
596 // Is the value an array?
597 
598             if (Object.prototype.toString.apply(value) === '[object Array]') {
599 
600 // The value is an array. Stringify every element. Use null as a placeholder
601 // for non-JSON values.
602 
603                 length = value.length;
604                 for (i = 0; i < length; i += 1) {
605                     partial[i] = str(i, value) || 'null';
606                 }
607 
608 // Join all of the elements together, separated with commas, and wrap them in
609 // brackets.
610 
611                 v = partial.length === 0
612                     ? '[]'
613                     : gap
614                     ? '[\n' + gap + partial.join(',\n' + gap) + '\n' + mind + ']'
615                     : '[' + partial.join(',') + ']';
616                 gap = mind;
617                 return v;
618             }
619 
620 // If the replacer is an array, use it to select the members to be stringified.
621 
622             if (rep && typeof rep === 'object') {
623                 length = rep.length;
624                 for (i = 0; i < length; i += 1) {
625                     if (typeof rep[i] === 'string') {
626                         k = rep[i];
627                         v = str(k, value);
628                         if (v) {
629                             partial.push(quote(k) + (gap ? ': ' : ':') + v);
630                         }
631                     }
632                 }
633             } else {
634 
635 // Otherwise, iterate through all of the keys in the object.
636 
637                 for (k in value) {
638                     if (Object.prototype.hasOwnProperty.call(value, k)) {
639                         v = str(k, value);
640                         if (v) {
641                             partial.push(quote(k) + (gap ? ': ' : ':') + v);
642                         }
643                     }
644                 }
645             }
646 
647 // Join all of the member texts together, separated with commas,
648 // and wrap them in braces.
649

651                 ? '{}'
652                 : gap
653                 ? '{\n' + gap + partial.join(',\n' + gap) + '\n' + mind + '}'
654                 : '{' + partial.join(',') + '}';
655             gap = mind;
656             return v;
657         }

664 
665 // The stringify method takes a value and an optional replacer, and an optional
666 // space parameter, and returns a JSON text. The replacer can be a function
667 // that can replace values, or an array of strings that will select the keys.
668 // A default replacer method can be provided. Use of the space parameter can
669 // produce text that is more easily readable.
670

673             indent = '';

676 // many spaces.

707 // If the JSON object does not yet have a parse method, give it one.
708 
709     if (typeof JSON.parse !== 'function') {
710         JSON.parse = function (text, reviver) {
711 
712 // The parse method takes a text and an optional reviver function, and returns
713 // a JavaScript value if the text is a valid JSON text.
714 
715             var j;
716 
717             function walk(holder, key) {
718 
719 // The walk method is used to recursively walk the resulting structure so
720 // that modifications can be made.
721 
722                 var k = 0, v = 0, value = holder[key];
723                 if (value && typeof value === 'object') {
724                     for (k in value) {
725                         if (Object.prototype.hasOwnProperty.call(value, k)) {
726                             v = walk(value, k);
727                             if (v !== undefined) {
728                                 value[k] = v;
729                             } else {
730                                 delete value[k];
731                             }
732                         }
733                     }
734                 }
735                 return reviver.call(holder, key, value);
736             }
737 
738 
739 // Parsing happens in four stages. In the first stage, we replace certain
740 // Unicode characters with escape sequences. JavaScript handles many characters
741 // incorrectly, either silently deleting them, or treating them as line endings.
742 
743             text = String(text);
744             cx.lastIndex = 0;
745             if (cx.test(text)) {
746                 text = text.replace(cx, function (a) {
747                     return '\\u' +
748                         ('0000' + a.charCodeAt(0).toString(16)).slice(-4);
749                 });
750             }
751 
752 // In the second stage, we run the text against regular expressions that look
753 // for non-JSON patterns. We are especially concerned with '()' and 'new'
754 // because they can cause invocation, and '=' because it can cause mutation.
755 // But just to be safe, we want to reject all unexpected forms.
756 
757 // We split the second stage into 4 regexp operations in order to work around
758 // crippling inefficiencies in IE's and Safari's regexp engines. First we
759 // replace the JSON backslash pairs with '@' (a non-JSON character). Second, we
760 // replace all simple value tokens with ']' characters. Third, we delete all
761 // open brackets that follow a colon or comma or that begin the text. Finally,
762 // we look to see that the remaining characters are only whitespace or ']' or
763 // ',' or ':' or '{' or '}'. If that is so, then the text is safe for eval.
764 
765             if (/^[\],:{}\s]*$/
766                     .test(text.replace(/\\(?:["\\\/bfnrt]|u[0-9a-fA-F]{4})/g, '@')
767                         .replace(/"[^"\\\n\r]*"|true|false|null|-?\d+(?:\.\d*)?(?:[eE][+\-]?\d+)?/g, ']')
768                         .replace(/(?:^|:|,)(?:\s*\[)+/g, ''))) {
769 
770 // In the third stage we use the eval function to compile the text into a
771 // JavaScript structure. The '{' operator is subject to a syntactic ambiguity
772 // in JavaScript: it can begin a block or an object literal. We wrap the text
773 // in parens to eliminate the ambiguity.
774 
775                 j = eval('(' + text + ')');
776 
777 // In the optional fourth stage, we recursively walk the new structure, passing
778 // each name/value pair to a reviver function for possible transformation.
779 
780                 return typeof reviver === 'function'
781                     ? walk({'': j}, '')
782                     : j;
783             }
784 
785 // If the text is not JSON parseable, then a SyntaxError is thrown.
786 
787             throw new SyntaxError('JSON.parse');
788         };

803 /* Creates a new queue. A queue is a first-in-first-out (FIFO) data structure -
804  * items are added to the end of the queue and removed from the front.
805  */
806 function Queue(){
807 
808   // initialise the queue and offset
809   var queue  = [];
810   var offset = 0;
811 
812   /* Returns the length of the queue.
813    */
814   this.getLength = function(){
815 
816     // return the length of the queue
817     return (queue.length - offset);
818 
819   };
820 
821   /* Returns true if the queue is empty, and false otherwise.
822    */
823   this.isEmpty = function(){
824 
825     // return whether the queue is empty
826     return (queue.length == 0);
827 
828   };
829 
830   /* Enqueues the specified item. The parameter is:
831    *
832    * item - the item to enqueue
833    */
834   this.enqueue = function(item){
835 
836     // enqueue the item
837     queue.push(item);
838 
839   };
840 
841   /* Dequeues an item and returns it. If the queue is empty then undefined is
842    * returned.
843    */
844   this.dequeue = function(){
845 
846     // if the queue is empty, return undefined
847     if (queue.length == 0) return undefined;
848 
849     // store the item at the front of the queue
850     var item = queue[offset];
851 
852     // increment the offset and remove the free space if necessary
853     if (++ offset * 2 >= queue.length){
854       queue  = queue.slice(offset);
855       offset = 0;
856     }
857 
858     // return the dequeued item
859     return item;
860 
861   };
862 
863   /* Returns the item at the front of the queue (without dequeuing it). If the
864    * queue is empty then undefined is returned.
865    */
866   this.peek = function(){
867 
868     // return the item at the front of the queue
869     return (queue.length > 0 ? queue[offset] : undefined);
870 
871   };
872 }
